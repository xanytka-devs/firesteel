cmake_minimum_required(VERSION 3.12)
project(XEngine)
set(PROJECT_NAME XEngine)

option(XE_BUILD_ENGINE "Build engine project." ON)
option(XE_BUILD_EDITOR "Build default editor config." ON)
option(XE_BUILD_EXAMPLES "Build other examples." ON)

if(XE_BUILD_ENGINE) 
	add_subdirectory(xengine)
endif()
if(XE_BUILD_EDITOR)
	add_subdirectory(examples/xeditor)
endif()
if(XE_BUILD_EXAMPLES)
	add_subdirectory(examples/triangle_demo)
	add_subdirectory(examples/audio_demo)
endif()

set_property(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT xeditor)
